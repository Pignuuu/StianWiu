Reaction Roles.js 

            
                
                

    // Event for giving respective role to user when reacting to respective reaction.
    // Role is also removed when removing their reaction


    // Get all required variables
    const Discord = require("discord.js");

    const CLIENT_KEY = "TOKEN";

    const client = new Discord.Client();

    const reactionChannel = "CHANNEL ID"

    const javascriptEmoji = "EMOJI ID"
    const pythonEmoji = "EMOJI ID"

    const javascriptRole = "ROLE ID"
    const pythonRole = "ROLE ID"

    const guildId = "SERVER ID"
    const channelId = "CHANNEL ID"
    const messageId = "MESSAGE ID"

    // Start bot(s)

    async function startbots() {
        console.log('\x1b[33m%s\x1b[0m', "Starting Client...");
        client.login(CLIENT_KEY);
    }
    startbots();

    // Find message inorder to cache it so bot can see reactions added

    client.on("ready", async() => {
        const guild = await client.guilds.fetch(guildId);
        const channel = guild.channels.cache.get(channelId);
        const message = await channel.messages.fetch(messageId);
    });

    // Add role when reaction is added

    client.on('messageReactionAdd', async(reaction, user) => {
        if (reaction.message.channel.id === reactionChannel) {
            const guild = await client.guilds.fetch(guildId);
            const channel = guild.channels.cache.get(channelId);
            const message = await channel.messages.fetch(messageId);

            if (reaction.emoji.id === javascriptEmoji) {
                const role = guild.roles.cache.find(role => role.id === javascriptRole);
                const member = guild.members.cache.find(member => member.id === user.id);
                member.roles.add(role);
            }
            if (reaction.emoji.id === pythonEmoji) {
                const role = guild.roles.cache.find(role => role.id === pythonRole);
                const member = guild.members.cache.find(member => member.id === user.id);
                member.roles.add(role);
            }
        }
    });

    // Remove role when reaction is removed

    client.on('messageReactionRemove', async(reaction, user) => {
        if (reaction.message.channel.id === reactionChannel) {
            const guild = await client.guilds.fetch(guildId);
            const channel = guild.channels.cache.get(channelId);
            const message = await channel.messages.fetch(messageId);

            if (reaction.emoji.id === javascriptEmoji) {
                const role = guild.roles.cache.find(role => role.id === javascriptRole);
                const member = guild.members.cache.find(member => member.id === user.id);
                member.roles.remove(role);
            }
            if (reaction.emoji.id === pythonEmoji) {
                const role = guild.roles.cache.find(role => role.id === pythonRole);
                const member = guild.members.cache.find(member => member.id === user.id);
                member.roles.remove(role);
            }
        }
    });

Slave & Master.js 

            
                
                

    // One main discord bot controls 3 or more other bots.
    // Nice for sending announcements for services like twitch & youtube etc...

    const Discord = require("discord.js");
    const util = require('util');
    const sleep = util.promisify(setTimeout);
    
    // Bot tokens
    const MASTER_KEY = "INSERT TOKEN";
    const SLAVE_KEY = "INSERT TOKEN";
    const SLAVE2_KEY = "INSERT TOKEN";
    const SLAVE3_KEY = "INSERT TOKEN";
    
    const master = new Discord.Client();
    const slave = new Discord.Client();
    const slave2 = new Discord.Client();
    const slave3 = new Discord.Client();
    
    // Connect discord bots, with 1500ms of delay to make discord happy :)
    
    async function startbots() {
        console.log("Starting Master...");
        master.login(MASTER_KEY);
        await sleep(1500);
        console.log("Starting slave...");
        slave.login(SLAVE_KEY)
        await sleep(1500);
        console.log("Starting Slave2...");
        slave2.login(SLAVE2_KEY)
        await sleep(1500);
        console.log("Starting Slave3...");
        slave3.login(SLAVE3_KEY)
    }
    startbots();
    
    master.on("ready", () => { console.log("Master connected!"); });
    slave.on("ready", () => { console.log("Slave connected!"); });
    slave2.on("ready", () => { console.log("Slave2 connected!"); });
    slave3.on("ready", () => { console.log("Slave3 connected!"); });
    
    // All commands
    
    function getParams(command) {
        const spcIndex = command.indexOf(" ");
        if (spcIndex > 0) {
            return command.slice(spcIndex, command.length);
        }
        return "";
    }
    
    // All slaves together
    
    master.on("message", async (message) => {
        if (message.content.toLowerCase().startsWith("/all")) {
    
            const subject = getParams(message.content);
            await message.channel.send(`Message sent to all slaves`);
    
            slave.channels.cache.get("INSERT CHANNEL ID").send(`${subject}`);
            await sleep(500);
            slave2.channels.cache.get("INSERT CHANNEL ID").send(`${subject}`);
            await sleep(500);
            slave3.channels.cache.get("INSERT CHANNEL ID").send(`${subject}`);
        }
    });
    
    // Seperate slave commands
    
    master.on("message", async (message) => {
        if (message.content.toLowerCase().startsWith("/1")) {
    
            const subject = getParams(message.content);
            await message.channel.send(`Message sent to slave`);
    
            slave.channels.cache.get("INSERT CHANNEL ID").send(`${subject}`);
    
        }
    });
    
    master.on("message", async (message) => {
        if (message.content.toLowerCase().startsWith("/2")) {
    
            const subject = getParams(message.content);
            await message.channel.send(`Message sent to slave2`);
    
            slave2.channels.cache.get("INSERT CHANNEL ID").send(`${subject}`);
        }
    });
    
    master.on("message", async (message) => {
        if (message.content.toLowerCase().startsWith("/3")) {
    
            const subject = getParams(message.content);
            await message.channel.send(`Message sent to slave3`);
    
            slave3.channels.cache.get("INSERT CHANNEL ID").send(`${subject}`);
        }
    });
